VALID SUDOKU import java.util.HashSet; class Solution { public boolean isValidSudoku(char[][] board) { for (int i = 0; i < board.length; i++) { HashSet rows = new HashSet<>(); for (int j = 0; j < board[i].length; j++) { char current = board[i][j]; if (current != '.') { if (!rows.add(current)) { return false; } } if(current != ".") { } } rows.clear(); } return true; // Return true if all checks pass } }
